## Usage

```tsx
import {
  ContextMenu,
  ContextMenuContent,
  ContextMenuItem,
  ContextMenuTrigger,
} from '@/components/ui/context-menu'
```

```tsx
<ContextMenu>
  <ContextMenuTrigger>Right-click on transaction</ContextMenuTrigger>
  <ContextMenuContent>
    <ContextMenuItem>View details</ContextMenuItem>
    <ContextMenuItem>Copy transaction ID</ContextMenuItem>
    <ContextMenuItem>View in ledger</ContextMenuItem>
    <ContextMenuItem variant="destructive">Revert transaction</ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```
