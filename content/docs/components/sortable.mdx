## Usage

```tsx
import {
  Sortable,
  SortableContent,
  SortableItem,
  SortableItemHandle,
  SortableOverlay,
} from '@/components/ui/sortable'
```

```tsx
<Sortable
  value={items}
  onValueChange={setItems}
  getItemValue={(item) => item.id}
>
  <SortableContent>
    {items.map((item) => (
      <SortableItem key={item.id} value={item.id}>
        <SortableItemHandle asChild>
          <button>Drag</button>
        </SortableItemHandle>
        {item.label}
      </SortableItem>
    ))}
  </SortableContent>
  <SortableOverlay />
</Sortable>
```
